KY-027 MAGIC LIGHT CUP MODULE

Краткое описание модуля:

"Arduino KY-027 Magic Light Cup Module" - представляет собой набор из двух плат, каждая из которых имеет светодиод и ртутный переключатель наклона. Используя PWM для управления светодиодами на каждом модуле, вы можете добиться эффекта «волшебного» перехода света от одного модуля к другому при их наклоне.

Схема подключения к Arduino:

Код программы для подключения:

  1. int ledPinA = 9;
  2. int switchPinA = 8;
  3. int switchStateA = 0;
  4. int ledPinB = 6;
  5. int switchPinB = 7;
  6. int switchStateB = 0;
  7. int brightness = 0;
  9. void setup()
  10. {
  11. pinMode(ledPinA, OUTPUT);
  12. pinMode(ledPinB, OUTPUT);
  13. pinMode(switchPinA, INPUT);
  14. pinMode(switchPinB, INPUT);
  15. }
  17. void loop()
  18. {
  19. switchStateA = digitalRead(switchPinA);
  20. if (switchStateA == HIGH && brightness != 255)
  21. {
  22. brightness ++;
  23. }
  24. switchStateB = digitalRead(switchPinB);
  25. if (switchStateB == HIGH && brightness != 0)
  26. {
  27. brightness --;
  28. }
  29. analogWrite(ledPinA, brightness); // A slow fade out
  30. analogWrite(ledPinB, 255 - brightness); // B slow bright up
  31. delay(20);
  32. }
