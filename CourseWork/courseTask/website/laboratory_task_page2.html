      Arduino Traffic Lights

      Состав устройства:

      1. Arduino
      2. 3 светодиода
      3. 3 резистора на 100 Ом
      4. 3 провода для подключения
      5. Доска для подключения
      Код подключения к модулю

      Код программы для подключения:

      1. // variables
      2. int GREEN = 2;
      3. int YELLOW = 3;
      4. int RED = 4;
      5. int DELAY_GREEN = 5000;
      6. int DELAY_YELLOW = 2000;
      7. int DELAY_RED = 5000;
      8. // basic functions
      9. void setup()
      10. {
      11. pinMode(GREEN, OUTPUT);
      12. pinMode(YELLOW, OUTPUT);
      13. pinMode(RED, OUTPUT);
      14. }
      15. void loop()
      16. {
      17. green_light();
      18. delay(DELAY_GREEN);
      19. yellow_light();
      20. delay(DELAY_YELLOW);
      21. red_light();
      22. delay(DELAY_RED);
      23. }
      24. void green_light()
      25. {
      26. digitalWrite(GREEN, HIGH);
      27. digitalWrite(YELLOW, LOW);
      28. digitalWrite(RED, LOW);
      29. }
      30. void yellow_light()
      31. {
      32. digitalWrite(GREEN, LOW);
      33. digitalWrite(YELLOW, HIGH);
      34. digitalWrite(RED, LOW);
      35. }
      36. void red_light()
      37. {
      38. digitalWrite(GREEN, LOW);
      39. digitalWrite(YELLOW, LOW);
      40. digitalWrite(RED, HIGH);
      41. }
      Просмотр реакции устройства

      Сценарий работы устройства:

      1. Загорается зеленый светодиод (задержка 5000 мс)
      2. Загорается желтый светодиод (задержка 2000 мс)
      3. Загорается красный светодиод (задержка 5000 мс)
